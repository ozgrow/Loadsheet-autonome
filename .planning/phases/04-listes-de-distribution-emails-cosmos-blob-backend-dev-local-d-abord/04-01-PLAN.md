# Backend pattern (api/send-email/index.js, à reproduire)

```javascript
// Auth header check — Azure SWA strips Authorization, use x-auth-token
function verifyToken(req) {
  var token = req.headers["x-auth-token"] || null;
  if (!token) {
    var auth = req.headers["authorization"] || "";
    token = auth.startsWith("Bearer ") ? auth.slice(7) : null;
  }
  if (!token) return null;
  try { return jwt.verify(token, process.env.JWT_SECRET); }
  catch (e) { return null; }
}

// Regex email + parsing "Name <addr@x.com>" — SOURCE DE VÉRITÉ
var emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
var angleBracketRegex = /<([^>]+)>/;
function validateEmails(str) {
  return String(str || '').split(/[,;]/).map(function(e) { return e.trim(); }).filter(Boolean).map(function(e) {
    var match = angleBracketRegex.exec(e);
    return match ? match[1].trim() : e;
  }).filter(function(e) {
    return emailRegex.test(e);
  });
}

// Réponse Function pattern
context.res = { status: 401, body: { error: "Non autorise." } };
```
