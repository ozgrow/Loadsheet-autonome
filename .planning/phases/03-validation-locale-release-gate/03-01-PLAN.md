```javascript
function isLoggedIn() {
  const session = sessionStorage.getItem(AUTH_SESSION_KEY);
  if (!session) return false;
  try {
    const { expiry } = JSON.parse(session);
    if (expiry <= Date.now()) return false;
    ```

    - `data === null` (JSON.parse("null")) → `!data` → return false
